{{- $url := "https://docs.google.com/spreadsheets/d/e/2PACX-1vQuJXdzcEO0f0WDSTHpAefUZQjErZ3aFnOeiRvCEB3xGYiemdD2klSGnnqHzLSyVA/pub?gid=1073340661&single=true&output=csv" -}} {{- /* 1. getCSV 대신 GetRemote를 사용하여 데이터를 가져옵니다 */ -}} {{- $res := resources.GetRemote $url -}} {{- $projects := slice -}} {{- with $res -}} {{- if .Err -}} {{- errorf "CSV 로딩 실패: %s" .Err -}} {{- else -}} {{- /* 2. 가져온 데이터를 CSV 형식으로 파싱합니다 */ -}} {{- $csv := .Content | transform.Unmarshal (dict "delimiter" ",") -}} {{- if gt (len $csv) 0 -}} {{- $keys := index $csv 0 -}} {{- $rows := after 1 $csv -}} {{- range $row := $rows -}} {{- $project := dict -}} {{- range $index, $value := $row -}} {{- $key := index $keys $index -}} {{- $project = merge $project ( dict $key $value ) -}} {{- end -}} {{- $projects = $projects | append $project -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- return $projects -}}
